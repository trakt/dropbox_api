def build_request(params, body)
  headers = {
    'Dropbox-API-Arg' => JSON.dump(params),
    'Content-Type' => 'application/octet-stream'
  }

  content_length = get_content_length body
  headers['Content-Length'] = content_length unless content_length.nil?

  return body, headers
end
